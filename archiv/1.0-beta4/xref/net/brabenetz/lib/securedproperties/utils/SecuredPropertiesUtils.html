20  package net.brabenetz.lib.securedproperties.utils;
21  
22  import com.github.fge.lambdas.Throwing;
23  import org.apache.commons.io.FileUtils;
24  import org.apache.commons.lang3.Validate;
25  import org.apache.commons.lang3.tuple.Pair;
26  
27  import java.io.ByteArrayInputStream;
28  import java.io.File;
29  import java.nio.charset.StandardCharsets;
30  import java.util.List;
31  import java.util.Properties;
32  import java.util.regex.Matcher;
33  import java.util.regex.Pattern;
34  import java.util.stream.Collectors;
35  
36  /**
37   * Some Utilities for {@link net.brabenetz.lib.securedproperties.SecuredProperties}.
38   */
39  public final class SecuredPropertiesUtils {
40  
41      private SecuredPropertiesUtils() {
42          super();
43      }
44  
45      /**
46       * Loads a {@link Properties} object from a given Property {@link File} and transform checked
47       * Exception into RuntimeExceptions.
48       *
49       * @param propertyFile
50       *        the property File
51       * @return the {@link Properties} object loaded by the propertyFile
52       */
53      public static Properties readProperties(final File propertyFile) {
54          final Properties properties = new Properties();
55          Validate.notNull(propertyFile, "The Propety-File is required");
56          byte[] propertyFileContent = Throwing.supplier(() -> FileUtils.readFileToByteArray(propertyFile)).get();
57          Throwing.runnable(() -> properties.load(new ByteArrayInputStream(propertyFileContent))).run();
58          return properties;
59      }
60  
61      /**
62       * Replaces the value for one key in the given Properties file and leaves all other lines unchanged.
63       *
64       * @param propertyFile
65       *        The property File with the given Key
66       * @param newProperties
67       *        The new Property, where left is the key, and right is the new value.
68       */
69      @SafeVarargs
70      public static void replaceSecretValue(final File propertyFile, final Pair<String, String>... newProperties) {
71          List<String> lines = Throwing.supplier(() -> FileUtils.readLines(propertyFile, StandardCharsets.ISO_8859_1.name())).get();
72          List<String> newLines = lines.stream()
73              .map((line) -> replaceValue(line, newProperties))
74              .collect(Collectors.toList());
75          Throwing.runnable(() -> FileUtils.writeLines(propertyFile, StandardCharsets.ISO_8859_1.name(), newLines)).run();
76      }
77  
78      // SuppressWarnings "PMD.DefaultPackage": only used in UnitTest
79      @SafeVarargs
80      @SuppressWarnings("PMD.DefaultPackage")
81      static String replaceValue(final String line, final Pair<String, String>... newProperties) {
82          for (Pair<String, String> newProperty : newProperties) {
83              String key = newProperty.getLeft();
84              String newValue = newProperty.getRight();
85              Pattern pattern = Pattern.compile("^" + Pattern.quote(key) + "(\\s*=\\s*).*$");
86              Matcher matcher = pattern.matcher(line);
87              if (matcher.matches()) {
88                  String gr1 = matcher.group(1);
89                  return key + gr1 + newValue;
90              }
91          }
92          return line;
93      }
