47      @Test
48      public void testToPreventCycleDependencies() throws Exception {
49          JDepend jdepend = new JDepend(PackageFilter.all().including(
50              SecuredProperties.class.getPackage().getName())
51              .excludingRest());
52          jdepend.addDirectory("./target/classes");
53          jdepend.analyze();
54          Assert.assertFalse(jdepend.containsCycles());
55  
56          JavaPackage pkgRoot = jdepend.getPackage(SecuredProperties.class.getPackage().getName());
57          JavaPackage pkgCore = jdepend.getPackage(Encryption.class.getPackage().getName());
58          JavaPackage pkgUtil = jdepend.getPackage(SecuredPropertiesUtils.class.getPackage().getName());
59  
60          // pkg uses other Pkg
61          assertThat(pkgRoot.getEfferents(), containsInAnyOrder(pkgUtil, pkgCore));
62          assertThat(pkgUtil.getEfferents(), is(empty()));
63          assertThat(pkgCore.getEfferents(), is(empty()));
64          // pkg is used in other Pkg
65          assertThat(pkgRoot.getAfferents(), is(empty()));
66          assertThat(pkgUtil.getAfferents(), containsInAnyOrder(pkgRoot));
67          assertThat(pkgCore.getAfferents(), containsInAnyOrder(pkgRoot));
68      }

70      @Test
71      public void testDependenciesOfPackages() throws Exception {
72  
73          String pkgBase = SecuredProperties.class.getPackage().getName();
74          Map<String, JavaPackage> rootPkgs = analyseDirectChildes(pkgBase);
75          assertNoCycleBetweenSiblings(rootPkgs);
76  
77          assertThat(rootPkgs.keySet(), containsInAnyOrder("core", "utils"));
78          assertThat(rootPkgs.get("core").getEfferents(), is(empty()));
79          assertThat(rootPkgs.get("utils").getEfferents(), is(empty()));
80  
81      }
82  
83      private void assertNoCycleBetweenSiblings(final Map<String, JavaPackage> rootPkgs) {
84          List<String> message = new ArrayList<>();
85          for (JavaPackage javaPackage : rootPkgs.values()) {
86              ArrayList<JavaPackage> cycles = new ArrayList<>();
87              boolean containsCycle = javaPackage.collectCycle(cycles);
88              if (containsCycle) {
89                  message.add(String.format("Package '%s' has cycles with siblings: %s", javaPackage.getName(), cycles));
90              }
91          }
92          Assert.assertTrue(message.stream().collect(Collectors.joining("\n")), message.isEmpty());
93      }

95      @SuppressWarnings("unused")
96      private JavaPackage[] toJavaPackage(final String parentPackage, final String... subPackages) {
97          return Arrays.stream(subPackages)
98              .map((subPackage) -> new JavaPackage(parentPackage + "." + subPackage))
99              .collect(Collectors.toList()).toArray(new JavaPackage[subPackages.length]);
100     }

102     private Map<String, JavaPackage> analyseDirectChildes(final String pkgRoot) throws Exception {
103 
104         JDepend jdepend = new JDepend(PackageFilter.all().including(pkgRoot).excludingRest());
105         jdepend.addDirectory("./target/classes");
106         jdepend.analyze();
107 
108         Map<String, JavaPackage> rootPkgs = getDirectChildPackages(jdepend, pkgRoot);
109 
110         fillDirectChildDependencies(jdepend, rootPkgs);
111 
112         return rootPkgs;
113     }
114 
115     private void fillDirectChildDependencies(final JDepend jdepend, final Map<String, JavaPackage> rootPkgs) {
116         if (rootPkgs.isEmpty()) {
117             // nothing to fill
118             return;
119         }
120 
121         String pkgRoot = StringUtils.substringBeforeLast(rootPkgs.values().iterator().next().getName(), ".");
122 
123         for (JavaPackage javaPackage : jdepend.getPackages()) {
124             String subPackage = getSubPackageName(pkgRoot, javaPackage);
125             if (StringUtils.isEmpty(subPackage)) {
126                 continue;
127             }
128             JavaPackage javaSubPackage = rootPkgs.get(subPackage);
129 
130             // System.out.println(String.format("%s: ", subPackage));
131             javaPackage.getEfferents()
132             .stream()
133             .map((pkg) -> getSubPackageName(pkgRoot, pkg))
134             .filter(StringUtils::isNotEmpty)
135             .map(rootPkgs::get)
136             .forEach(javaSubPackage::addEfferent);
137 
138             javaPackage.getAfferents()
139             .stream()
140             .map((pkg) -> getSubPackageName(pkgRoot, pkg))
141             .filter(StringUtils::isNotEmpty)
142             .map(rootPkgs::get)
143             .forEach(javaSubPackage::addAfferent);
144 
145             javaPackage.getClasses().forEach(javaSubPackage::addClass);
146 
147         }
148     }
149 
150     private Map<String, JavaPackage> getDirectChildPackages(final JDepend jdepend, final String pkgRoot) {
151         Map<String, JavaPackage> rootPkgs = new HashMap<>();
152         for (JavaPackage javaPackage : jdepend.getPackages()) {
153             String subPackage = getSubPackageName(pkgRoot, javaPackage);
154             if (StringUtils.isEmpty(subPackage)) {
155                 continue;
156             }
157             JavaPackage javaSubPackage = rootPkgs.get(subPackage);
158             if (javaSubPackage == null) {
159                 javaSubPackage = new JavaPackage(pkgRoot + "." + subPackage);
160                 rootPkgs.put(subPackage, javaSubPackage);
161             }
162 
163         }
164         return rootPkgs;
165     }
166 
167     private String getSubPackageName(final String pkgRoot, final JavaPackage javaPackage) {
168         String subPackage = StringUtils.substringAfter(javaPackage.getName(), pkgRoot + ".");
169         subPackage = StringUtils.substringBefore(subPackage, ".");
170         return subPackage;
171     }
